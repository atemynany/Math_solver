\begin{theorem}[Newton's Second Law]
For a particle of mass $m > 0$ subject to force $\vb{F}$ with acceleration $\vb{a}$:
\[
\vb{F} = m \vb{a}
\]
\end{theorem}

\section{Lean Formalization}

\begin{lstlisting}
theorem newton_second_law
    (m : ℝ) (a : ℝ → Vec3) (F : ℝ → Vec3)
    (hm : m > 0)
    (h : ∀ t, F t = m • a t) :
    ∀ t, (1 / m) • F t = a t := by
  intro t
  rw [h t]
  rw [smul_smul]
  simp [ne_of_gt hm]
\end{lstlisting}
